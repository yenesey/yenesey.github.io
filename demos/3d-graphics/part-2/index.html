function rotate(verts, a, b, t) {
  var vertsRotated = new Array(verts.length)

  for (var i = 0; i < verts.length; i++) {
    var x = verts[i].x
    var y = verts[i].y
    var z = verts[i].z

    var x1, y1, z1
    //    YZ   x
    y1 = y * cos(a) - z * sin(a)
    z1 = y * sin(a) + z * cos(a)
    y = y1
    z = z1

    //    XZ   y
    z1 = z * cos(b) - x * sin(b)
    x1 = z * sin(b) + x * cos(b)
    x = x1
    z = z1

    //    XY   z
    x1 = x * cos(t) - y * sin(t)
    y1 = x * sin(t) + y * cos(t)
    x = x1
    y = y1

    vertsRotated[i] = new Vertex(x, y, z)
  }
  return vertsRotated
}
